112// Conversions between SL_Bus_DOB_ros_std_msgs_Header and std_msgs::Header



113



114void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_DOB_ros_std_msgs_Header const* busPtr)



115{



116  const std::string rosMessageType("std_msgs/Header");



117



118  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);



119  msgPtr->seq =  busPtr->Seq;



120  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



121}



122



123void convertToBus(SL_Bus_DOB_ros_std_msgs_Header* busPtr, std_msgs::Header const* msgPtr)



124{



125  const std::string rosMessageType("std_msgs/Header");



126



127  convertToBusVariablePrimitiveArray(busPtr->FrameId, busPtr->FrameId_SL_Info, msgPtr->frame_id, slros::EnabledWarning(rosMessageType, "frame_id"));



128  busPtr->Seq =  msgPtr->seq;



129  convertToBus(&busPtr->Stamp, &msgPtr->stamp);



130}
