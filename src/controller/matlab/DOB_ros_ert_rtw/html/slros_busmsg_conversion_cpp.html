114void convertFromBus(std_msgs::Header* msgPtr, SL_Bus_DOB_ros_std_msgs_Header const* busPtr)



115{



116  const std::string rosMessageType("std_msgs/Header");



117



118  convertFromBusVariablePrimitiveArray(msgPtr->frame_id, busPtr->FrameId, busPtr->FrameId_SL_Info);

120  convertFromBus(&msgPtr->stamp, &busPtr->Stamp);



121}
